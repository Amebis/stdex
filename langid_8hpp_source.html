82 for (i = 0; ; ++i) {
83 auto a = stdex::tolower(str1[i]);
84 auto b = stdex::tolower(str2[i]);
85 auto a_end = !a || stdex::ispunct(a);
86 auto b_end = !b || stdex::ispunct(b);

89 if (a_end || a < b) return true;
